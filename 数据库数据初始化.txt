#数据库中的stu_sco表
DROP PROCEDURE IF EXISTS proc_initDatas;
DELIMITER $
CREATE PROCEDURE proc_initDatas()
BEGIN
    DECLARE i INT DEFAULT 201601001;
    WHILE i<=201601010 DO
        INSERT INTO stu_sco(stuno,courseid,score,type) VALUES(i,'c001', floor(rand()*100),'已批改');
        SET i = i+1;
    END WHILE;
END $
CALL proc_initDatas();
